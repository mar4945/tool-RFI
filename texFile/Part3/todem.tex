		\begin{align}

			&\overline{\phi}^i\left(t,t_0,x_0^i,u^i\right) \geq \phi^i\left(t,t_0,x_0^i,u^i,p^i\right), \label{lemma:rupProxy}\\
			&\forall p^i \in P^i, \; \forall x_0^i \in X^i,\; \forall u^i \in U^i\left(x^i\right), \forall t \geq t_0, \nonumber
		\end{align}
	\end{subequations}
